def self.parse(chunk)
  line = chunk.strip
  # thanks Net::HTTPResponse
  return [nil,nil] if chunk =~ /\AHTTP(?:\/(\d+\.\d+))?\s+(\d\d\d)\s*(.*)\z/in
  m = /\A([^:]+):\s*/.match(line)
  [m[1], m.post_match] rescue [nil, nil]
end
