def all
  if block_given?
    bucket.keys do |keys|
      keys.each do |key|
        obj = find_one(key)
        yield obj if obj
      end
    end
    []
  else
    bucket.keys.inject([]) do |acc, k|
      obj = find_one(k)
      obj ? acc << obj : acc
    end
  end
end

def find(*args)
  args.flatten!
  return nil if args.empty? || args.all?(&:blank?)
  return find_one(args.first) if args.one?
  args.map {|key| find_one(key) }
end

def find!(*args)
  found = find(*args)
  raise DocumentNotFound.new(args, found) if !found || Array(found).include?(nil)
  found
end
