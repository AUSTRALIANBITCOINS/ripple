def all
  if block_given?
    bucket.keys do |keys|
      keys.each do |key|
        obj = find_one(key)
        yield obj if obj
      end
    end
    []
  else
    bucket.keys.inject([]) do |acc, k|
      obj = find_one(k)
      obj ? acc << obj : acc
    end
  end
end

def find(*args)
  args.flatten!
  return [] if args.length == 0
  return find_one(args.first) if args.length == 1
  args.map {|key| find_one(key) }
end
