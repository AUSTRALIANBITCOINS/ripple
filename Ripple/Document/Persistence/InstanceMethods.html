def reload
  return self if new?
  robject.reload(:force => true)
  @attributes.merge!(@robject.data)
  self
end

def save
  robject.key = key if robject.key != key
  robject.data = attributes_for_persistence
  robject.store
  self.key = robject.key
  @new = false
  true
rescue Riak::FailedRequest
  false
end
