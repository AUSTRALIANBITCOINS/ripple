def method_missing(method, *args, &block)
  load_target

  if block_given?
    target.send(method, *args)  { |*block_args| block.call(*block_args) }
  else
    target.send(method, *args)
  end
end

def send(method, *args, &block)
  if proxy_respond_to?(method)
    super
  else
    load_target
    target.send(method, *args, &block)
  end
end
