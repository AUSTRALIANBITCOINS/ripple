def replace(value)
  @reflection.verify_type!(value, @owner)
  @owner.robject.links -= links
  Array(value).compact.each do |doc|
    doc.save if doc.new?
    @owner.robject.links << doc.robject.to_link(@reflection.link_tag)
  end
  loaded
  @target = value
end
