def default
  default = options[:default]

  return nil if default.nil?
  type_cast(default.respond_to?(:call) ? default.call : default)
end

def type_cast(value)
  if @type.respond_to?(:ripple_cast)
    @type.ripple_cast(value)
  else
    value
  end
end
